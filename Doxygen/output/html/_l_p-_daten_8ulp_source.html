269 void UpdateBBoxWire(UL_WIRE W, int layer) {
270  if (W.layer == layer) {
271  real w2 = W.width/2;
272  real xmin = min(W.x1 - w2, W.x2 - w2), ymin = min(W.y1 - w2, W.y2 - w2);
273  real xmax = max(W.x1 + w2, W.x2 + w2), ymax = max(W.y1 + w2, W.y2 + w2);
274  if (W.arc)
275  if (W.arc.angle2 > 360)
276  xmax = W.arc.xc + W.arc.radius + w2;
277  else if (((W.arc.angle1 < 90) && (W.arc.angle2 > 90)) || (W.arc.angle2 > 450))
278  ymax = W.arc.yc + W.arc.radius + w2;
279  else if (((W.arc.angle1 < 180) && (W.arc.angle2 > 180)) || (W.arc.angle2 > 540))
280  xmin = W.arc.xc - W.arc.radius - w2;
281  else if (((W.arc.angle1 < 270) && (W.arc.angle2 > 270)) || (W.arc.angle2 > 630))
282  ymin = W.arc.yc - W.arc.radius - w2;
283  UpdateBBoxBox(xmin, ymin, xmax, ymax);
284  }

396 void GetDruParams() {
397  string lines[];
398  int nrlines = fileread(lines, DruFile);
399  if (nrlines == 0) dlgMessageBox("Error reading design rules !");
400  int layer_used[], nrlayers, bottom_idx;
401 
402  for (int i = 0; i < nrlines; ++i) {
403  string words[];
404  int nrwords = strsplit(words, lines[i], ' ');
405  if (nrwords < 3) continue; // Empty line or some other stuff
406  string keyword = words[0];
407  if (keyword == "layerSetup")
408  for (int k = 1; k <= 16; ++k) {
409  string nr_pattern;
410  sprintf(nr_pattern, "[\\D]%d[\\D]", k); // Number, surrounded by 'no number character'
411  if (strxstr(words[2], nr_pattern) != -1) {
412  layer_used[k - 1] = true;
413  ++NrLayers;
414  bottom_idx = k - 1;
415  }
416  }
417  else if (keyword == "mtCopper") {
418  if (nrwords != 18) dlgMessageBox("Unexpected design rules !");
419  int top_thickness = -1, bot_thickness;
420  for (int k = 2; k < nrwords; ++k) {
421  if (layer_used[k - 2]) {
422  int thickness = Str2U(words[k]);
423  if (top_thickness == -1)
424  top_thickness = thickness;
425  else if (k - 2 == bottom_idx)
426  bot_thickness = thickness;
427  else
428  CopperThicknessInside = min(thickness, CopperThicknessInside);
429  MaterialThickness += thickness;
430  }
431  }
432  CopperThicknessOutside = min(top_thickness, bot_thickness);
433  }
434  else if (keyword == "mtIsolate") {
435  if (nrwords != 17) dlgMessageBox("Unexpected design rules !");
436  // n used layers => n-1 isolate layers. If only 1 layer, also 1 isolate layer.
437  for (int k = 2; k < nrwords; ++k)
438  if (layer_used[k - 2])
439  MaterialThickness += Str2U(words[k]);
440  }
441  else if (keyword == "mdSmdSmd")
442  MinPitch = Str2U(words[2]);
443  }
444 }

508 void CheckHoles() {
509  int idx[], start[], end[], k;
510  board(B) {
511  B.holes(H) {
512  MinHoleSize = min(MinHoleSize, H.drill);
513  AnzahlBohrduchmesser(H.drill);
514  }
515 
516  B.signals(S) S.vias(V) {
517  MinHoleSize = min(MinHoleSize, V.drill);
518  AnzahlBohrduchmesser(V.drill);
519  // Collect blind/buried holes !
520  if ((V.start > 1) && (V.end < 16)) {
521  start[k] = V.start;
522  end[k] = V.end;
523  ++k;
524  ++NrburiedVia;
525  } else if ( (V.start == 1 && V.end != 16) || (V.start != 1 && V.end == 16)) {
526  ++k;
527  ++NrblindVia;
528  }
529  }
530  B.elements(E) {
531  E.package.contacts(C) if (C.pad) {
532  //Debug(U2MmStr(C.pad.drill));
533  MinHoleSize = min(MinHoleSize, C.pad.drill);
534  AnzahlBohrduchmesser(C.pad.drill);
535  }
536  E.package.holes(H) {
537  //Debug(U2MmStr(H.drill));
538  MinHoleSize = min(MinHoleSize, H.drill);
539  AnzahlBohrduchmesser(H.drill);
540  }
541  }
542  }
543  // NrBlindBuriedTypes: Find out how many different ones exist !
544  sort(k, idx, start, end);
545  for (int l = 0, cur_s = 0, cur_e = 0; l < k; ++l)
546  if ((start[idx[l]] != cur_s) || (end[idx[l]] != cur_e)) {
547  ++NrBlindBuriedTypes;
548  cur_s = start[idx[l]];
549  cur_e = end[idx[l]];
550  }
551 
552  //Debug("Verschiedene Bohrdurchmesser\n" + debugkram);
553 }

555 void GetPCBParams() {
556  // Check out layers:
557  int smask_top, smask_bottom, silk_top, silk_bottom;
558  int nr_used_layers;
559  board(B) {

563  if (L.number <= 16) ++nr_used_layers;
564  else if (L.number == LAYER_TSTOP) smask_top = 1;
565  else if (L.number == LAYER_BSTOP) smask_bottom = 1;
566  else if (L.number == LAYER_TPLACE) silk_top = 1; // Could also take TNAMES/TVALUES. Leave the criterium less strict
567  else if (L.number == LAYER_BPLACE) silk_bottom = 1; // Could also take BNAMES/BVALUES. Leave the criterium less strict
568  }
569  SolderSides = smask_top ? (smask_bottom ? "Beidseitig" : "Oberseite") :
570  (smask_bottom ? "Unterseite" : "keine");
571  SilkScreenSides = (silk_top) ? (silk_bottom ? "Beidseitig" : "Oberseite") :
